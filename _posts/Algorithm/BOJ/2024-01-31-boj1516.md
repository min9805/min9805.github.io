from collections import deque

N = int(input())

dic = {}
for i in range(1, N + 1):
    dic[i] = []
indegree = [0] * (N + 1)
cost = [0] * (N + 1)
dp = [0] * (N + 1)

for i in range(1, N + 1):
    temp = input().split()
    dp[i] = cost[i] = int(temp[0])
    indegree[i] = len(temp[1:-1])
    for n in temp[1:-1]:
        dic[int(n)].append(i)

queue = deque()

for i in range(1, N + 1):
    if indegree[i] == 0:
        queue.append(i)

while queue:
    node = queue.popleft()
    for n in dic[node]:
        n = int(n)
        indegree[n] -= 1
        if indegree[n] == 0:
            queue.append(n)
        dp[n] = max(dp[n], cost[n] + dp[node])

for node in dp[1:]:
    print(node)
