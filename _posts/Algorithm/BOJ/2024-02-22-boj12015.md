import bisect

N = int(input())
arr = list(map(int, input().split()))
arr.insert(0, 0)
dp = [0]

for i in range(1, N + 1):
    index = bisect.bisect_left(dp, arr[i])
    if index == len(dp):
        dp.append(arr[i])
    else:
        dp[index] = arr[i]

print(len(dp) - 1)
