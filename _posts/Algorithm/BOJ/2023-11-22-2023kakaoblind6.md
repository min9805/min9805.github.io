def solution(n, m, x, y, r, c, k):
    dx = [-1, 0, 0, 1]
    dy = [0, 1, -1, 0]
    dpath = ['u', 'r', 'l', 'd']

    def get_dist(x1, y1, x2, y2):
        return abs(x1 - x2) + abs(y1 - y2)

    def dfs():
        stack = []
        stack.append([x, y, 0, ""])

        min_path = "z"

        while stack:
            cx, cy, cnt, path = stack.pop()
            for i in range(4):
                nx = cx + dx[i]
                ny = cy + dy[i]
                npath = path + dpath[i]
                ncnt = cnt + 1

                if nx <= 0 or ny <= 0 or nx > n or ny > m or ncnt + get_dist(nx, ny, r, c) > k or (k - ncnt - get_dist(nx, ny, r, c)) % 2 == 1:
                    continue

                ncnt = cnt + 1

                if ncnt == k and nx == r and ny == c:
                    return npath
                elif ncnt < k:
                    stack.append([nx, ny, ncnt, npath])

        return min_path

    answer = dfs()

    if answer == 'z':
        return "impossible"

    return answer
