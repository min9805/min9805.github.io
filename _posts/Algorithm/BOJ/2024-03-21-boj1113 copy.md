def DFS(h):
    visited = [[False for _ in range(M)] for _ in range(N)]

    # ê°€ì¥ìë¦¬ ë¶€ë¶„ì— h ë³´ë‹¤ ì‘ê±°ë‚˜ ê°™ì€ ê²½ìš°ì— stack ì— ì¶”ê°€.
    stack = []
    for i in range(N):
        for j in range(M):
            if i == 0 or j == 0 or i == N - 1 or j == M - 1:
                if pool[i][j] <= h:
                    stack.append((i, j))

    while stack:
        x, y = stack.pop()
        visited[x][y] = True

        for i in range(4):
            nx = x + dx[i]
            ny = y + dy[i]

            if nx < 0 or ny < 0 or nx >= N or ny >= M or visited[nx][ny]:
                continue

            if pool[nx][ny] >= h:
                visited[nx][ny] = True
            else:
                stack.append((nx, ny))

    c = 0
    for li in visited:
        c += sum(li)

    return (M * N) - c
