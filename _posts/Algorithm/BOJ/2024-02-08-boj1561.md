    heapq.heappush(attractions, (0, i + 1))

            end = mid
        else:
            start = mid + 1

    temp_sum = 0
    for t in time:
        temp_sum += temp_time // t + 1
    N -= temp_sum

            N -= 1
            if N == 0:
                print(i + 1)
