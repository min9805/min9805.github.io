board = [list(map(int, input().split())) for _ in range(10)]
max_cnt = 25
paper = [5, 5, 5, 5, 5]


def dfs(x, y, c):
    global max_cnt, paper

    if x >= 10:
        max_cnt = min(max_cnt, c)
        return max_cnt

    if y >= 10:
        dfs(x + 1, 0, c)
        return

    if board[x][y] == 1:
        for i in range(5):
            if paper[i] == 0:
                continue
            if y + i >= 10 or x + i >= 10:
                continue

            if not cal_paper(i, x, y):
                break

            for j in range(i + 1):
                for k in range(i + 1):
                    board[x + j][y + k] = 0

            paper[i] -= 1
            dfs(x, y + i + 1, c + 1)

            paper[i] += 1
            for j in range(i + 1):
                for k in range(i + 1):
                    board[x + j][y + k] = 1
    else:
        dfs(x, y + 1, c)


def cal_paper(i, x, y):
    for j in range(i + 1):
        for k in range(i + 1):
            if board[x + j][y + k] != 1:
                return False
    return True


dfs(0, 0, 0)
print(-1 if max_cnt == 25 else max_cnt)
