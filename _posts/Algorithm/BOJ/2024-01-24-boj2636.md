n, m = map(int, input().split())

dx = [0, 1, 0, -1]
dy = [1, 0, -1, 0]

matrix = []
for _ in range(n):
    matrix.append(list(map(int, input().split())))

target = set()
last_target = 0
flag = True
cnt = 0

while flag:
    visited = [[False for _ in range(m)] for _ in range(n)]

    stack = []
    stack.append([0, 0])
    while stack:
        i, j = stack.pop()

        for k in range(4):
            nx, ny = i + dx[k], j + dy[k]

            if nx < 0 or ny < 0 or nx >= n or ny >= m or visited[nx][ny] == True:
                continue

            visited[nx][ny] = True
            if matrix[nx][ny] == 0:
                stack.append([nx, ny])
            else:
                target.add((nx, ny))

    if not target:
        flag = False
        break

    for x, y in target:
        matrix[x][y] = 0

    cnt += 1
    last_target = len(target)
    target.clear()

print(cnt)
print(last_target)
