```
def solution(n, computers):
    answer = 0
    visited = [False for _ in range(n)]

    for i in range(n):
        if visited[i]:
            continue

        answer += 1

        stack = []
        stack.append(i)
        visited[i] = True
        while stack:
            node = stack.pop()
            for j in range(len(computers[node])):
                if computers[node][j] == 1 and not visited[j]:
                    stack.append(j)
                    visited[j] = True

    return answer
```
